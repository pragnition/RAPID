```javascript
async function handleMerge(cwd, subcommand, args) {
  const merge = require('../lib/merge.cjs');
  const worktree = require('../lib/worktree.cjs');

  switch (subcommand) {
    case 'review': {
      // args[0] = setName
      const setName = args[0];
      if (!setName) { error('Usage: merge review <set-name>'); process.exit(1); }
      const result = merge.runProgrammaticGate(cwd, setName);
      const setDir = path.join(cwd, '.planning', 'sets', setName);
      // Write initial REVIEW.md with programmatic results only (agent review added by skill)
      merge.writeReviewMd(setDir, {
        setName,
        verdict: result.passed ? 'PENDING_REVIEW' : 'BLOCK',
        contractResults: { valid: result.contractValid },
        ownershipResults: { violations: result.ownershipViolations },
        testResults: { pass: result.testsPass, output: result.testOutput },
        findings: { blocking: result.ownershipViolations.map(v => `Ownership: ${v.file} owned by ${v.owner}`), fixable: [], suggestions: [] },
      });
      output(JSON.stringify(result));
      break;
    }

    case 'execute': {
      const setName = args[0];
      if (!setName) { error('Usage: merge execute <set-name>'); process.exit(1); }
      const baseBranch = worktree.detectMainBranch(cwd);
      const result = merge.mergeSet(cwd, setName, baseBranch);
      if (result.merged) {
        // Update registry with merge status
        await worktree.registryUpdate(cwd, (reg) => {
          if (reg.worktrees[setName]) {
            reg.worktrees[setName].mergeStatus = 'merged';
            reg.worktrees[setName].mergedAt = new Date().toISOString();
            reg.worktrees[setName].mergeCommit = result.commitHash;
          }
          return reg;
        });
      }
      output(JSON.stringify(result));
      break;
    }

    case 'status': {
      const registry = worktree.loadRegistry(cwd);
      const statuses = {};
      for (const [name, wt] of Object.entries(registry.worktrees || {})) {
        statuses[name] = {
          phase: wt.phase || 'unknown',
          mergeStatus: wt.mergeStatus || 'pending',
          mergedAt: wt.mergedAt || null,
          mergeCommit: wt.mergeCommit || null,
        };
      }
      output(JSON.stringify(statuses));
      break;
    }

    case 'integration-test': {
      const result = merge.runIntegrationTests(cwd);
      output(JSON.stringify(result));
      break;
    }

    case 'order': {
      const order = merge.getMergeOrder(cwd);
      output(JSON.stringify(order));
      break;
    }

    case 'update-status': {
      const setName = args[0];
      const status = args[1];
      if (!setName || !status) { error('Usage: merge update-status <set> <status>'); process.exit(1); }
      await worktree.registryUpdate(cwd, (reg) => {
        if (reg.worktrees[setName]) {
          reg.worktrees[setName].mergeStatus = status;
        }
        return reg;
      });
      output(JSON.stringify({ updated: true, set: setName, mergeStatus: status }));
      break;
    }

    default:
      error(`Unknown merge subcommand: ${subcommand}`);
      process.exit(1);
  }
}
```

```markdown
   \`\`\`bash
   node ~/RAPID/rapid/src/bin/rapid-tools.cjs merge update-status {setName} reviewing
   \`\`\`

     > Detail: {detail}

Display progress:
> [{waveNum}/{totalWaves}] {setName}: MERGED (commit {commitHash})
```
