    3. Add the `handleDisplay` function (place it near the bottom, before `main()`):
       ```javascript
       function handleDisplay(subcommand, args) {
         const { renderBanner } = require('../lib/display.cjs');

         switch (subcommand) {
           case 'banner': {
             const stage = args[0];
             if (!stage) {
               error('Usage: rapid-tools display banner <stage> [target]');
               process.exit(1);
             }
             const target = args.slice(1).join(' ');
             // Banner outputs raw formatted text, NOT JSON
             // This is intentional -- banners are visual output, not data
             process.stdout.write(renderBanner(stage, target) + '\n');
             break;
           }
           default:
             error(`Unknown display subcommand: ${subcommand}`);
             process.exit(1);
         }
       }
       ```

    IMPORTANT: The display command is the ONLY rapid-tools command that outputs raw formatted text instead of JSON. This is intentional per CONTEXT.md. Skills should call it as a standalone command and NOT capture its output with $(...) syntax.
