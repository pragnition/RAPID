function generateFrontmatter(role) {
  const tools = ROLE_TOOLS[role] || 'Read, Bash, Grep, Glob';
  const description = ROLE_DESCRIPTIONS[role] || `RAPID ${role} agent`;
  return `---
name: rapid-${role}
