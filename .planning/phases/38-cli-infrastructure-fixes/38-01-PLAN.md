         ```javascript
         function parseQuickAddArgs(args) {
           let commitHash = null;
           let directory = null;
           const descParts = [];
           for (let i = 0; i < args.length; i++) {
             if (args[i] === '--commit' && i + 1 < args.length) {
               commitHash = args[++i];
             } else if (args[i] === '--dir' && i + 1 < args.length) {
               directory = args[++i];
             } else {
               descParts.push(args[i]);
             }
           }
           return { description: descParts.join(' '), commitHash, directory };
         }
         ```
