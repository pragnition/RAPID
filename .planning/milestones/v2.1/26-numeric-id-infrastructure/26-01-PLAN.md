/**
 * List all set names from .planning/sets/ directory.
 * Filters out non-directory entries (DAG.json, OWNERSHIP.json, etc.).
 * @param {string} cwd - Project root directory
 * @returns {string[]} Sorted array of set directory names
 */
function listSets(cwd) {
  const setsDir = path.join(cwd, '.planning', 'sets');
  if (!fs.existsSync(setsDir)) { return []; }
  const entries = fs.readdirSync(setsDir, { withFileTypes: true });
  return entries.filter((entry) => entry.isDirectory()).map((entry) => entry.name).sort();
}

<feature>
  <name>Numeric ID Resolver (resolveSet + resolveWave)</name>
  <files>src/lib/resolve.cjs, src/lib/resolve.test.cjs, src/bin/rapid-tools.cjs</files>
  <behavior>
    ## resolveSet(input, cwd) -- UX-01 + UX-03

    Numeric resolution (UX-01):
    - resolveSet("1", cwd) with 3 sets ["set-01-api", "set-02-data", "set-03-ui"] -> { resolvedId: "set-01-api", numericIndex: 1, wasNumeric: true }
    - resolveSet("3", cwd) -> { resolvedId: "set-03-ui", numericIndex: 3, wasNumeric: true }

    Backward compat (UX-03):
    - resolveSet("set-02-data", cwd) -> { resolvedId: "set-02-data", numericIndex: 2, wasNumeric: false }

    Error cases:
    - resolveSet("0", cwd) -> throws "Invalid index: must be a positive integer."
    - resolveSet("-1", cwd) -> "-1" does NOT match /^\d+$/, so treated as string ID lookup, throws "Set '-1' not found."
    - resolveSet("99", cwd) with 3 sets -> throws "Set 99 not found. Valid range: 1-3. Use /rapid:status to see available sets."
    - resolveSet("1", cwd) with no sets -> throws "No sets found. Run /rapid:plan first to create a project plan with sets."
    - resolveSet("nonexistent", cwd) -> throws "Set 'nonexistent' not found. Available sets: set-01-api, set-02-data, set-03-ui"

    Boundary:
    - resolveSet("1", cwd) with 1 set -> returns that set with numericIndex: 1 (first and last)

    ## resolveWave(input, cwd) -- UX-02 + UX-03

    Design: resolveWave accepts `state` as second param (sync, testable). The CLI handler reads state and passes it in. For numeric input, resolveWave also needs cwd (for set resolution via listSets).
    Signature: `resolveWave(input, state, cwd)` -- state is the parsed ProjectState, cwd is for set directory lookup.

    Numeric resolution (UX-02):
    - resolveWave("1.1", state, cwd) with set 1 having 2 waves -> { setId: "set-01-api", waveId: "wave-01", setIndex: 1, waveIndex: 1, wasNumeric: true }
    - resolveWave("2.3", state, cwd) with set 2 having 3 waves -> { setId: "set-02-data", waveId: "wave-03", setIndex: 2, waveIndex: 3, wasNumeric: true }

    Backward compat (UX-03) -- string wave ID:
    - resolveWave("wave-01", state, cwd) -> delegates to wave-planning.resolveWave(state, "wave-01"), wraps result adding setIndex/waveIndex/wasNumeric:false

    Error cases:
    - resolveWave("1.", state, cwd) -> throws "Invalid wave reference. Use N.N format (e.g., 1.1 = set 1, wave 1)."
    - resolveWave(".1", state, cwd) -> throws same malformed error
    - resolveWave("1.0", state, cwd) -> throws "Invalid index: must be a positive integer." (wave index 0)
    - resolveWave("0.1", state, cwd) -> throws "Invalid index: must be a positive integer." (set index 0)
    - resolveWave("1.1.1", state, cwd) -> does NOT match /^\d+\.\d+$/, treated as string, throws wave not found
    - resolveWave("1.99", state, cwd) where set 1 has 2 waves -> throws "Wave 99 not found in set 'set-01-api'. Valid range: 1-2."
    - resolveWave("99.1", state, cwd) -> throws set out of range error

    ## CLI: rapid-tools resolve set|wave <input>

    - `rapid-tools resolve set 1` -> outputs JSON { resolvedId, numericIndex, wasNumeric }
    - `rapid-tools resolve wave 1.1` -> outputs JSON { setId, waveId, setIndex, waveIndex, wasNumeric }
    - `rapid-tools resolve set 99` -> outputs JSON { error: "..." } and exits 1
    - `rapid-tools resolve wave 1.` -> outputs JSON { error: "..." } and exits 1
    - `rapid-tools resolve foo 1` -> exits 1 with usage error
    - `rapid-tools resolve` with no subcommand -> exits 1 with usage error
  </behavior>
